void reverse(char *left, char *right)
{
	while(left < right)
	{
		char tmp = *left;
		*left = *right;
		*right = tmp;
		left++;
		right--;
	}
}
void left_move(char *arr, int k)  ////旋转字符  三步旋转法
{
	int len = strlen(arr);
	reverse(arr, arr+k-1);
	reverse(arr+k, arr+len-1);
	reverse(arr, arr+len-1);
}
